single process - parse output

  • shlex (shell lexical analyzer) has method split() that parses string into list
  • subprocess.run() needs in first argument parsed list (provided by shlex.split()
  • good for single process execution, shell pipes don’t work
import subprocess
import shlex
import pathlib

cmd = 'git ls-tree --full-tree -r --name-only HEAD'

out = subprocess.run(shlex.split(cmd),
    check=True,
    encoding="utf-8",
    universal_newlines=True,
    stdout=subprocess.PIPE).stdout

for line in out.split('\n'):
    p = pathlib.PosixPath(f"/data/notes/{line}")
    if p.is_file():
        print(line)

logging

import logging
import sys

# logger init
log_format = '%(asctime)s:%(levelname)s:%(name)s:%(funcName)s:%(lineno)d:%(message)s'
if "-vv" in sys.argv[1:]: logging_level = logging.DEBUG
elif "-v" in sys.argv[1:]: logging_level = logging.INFO
else: logging_level = logging.WARNING
logging.basicConfig(format=log_format, level=logging_level)

# using logger
logging.debug('This is a debug message')
logging.info('This is an info message')
logging.warning('This is a warning message')
logging.error('This is an error message')
logging.critical('This is a critical message')
